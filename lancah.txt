#include <WiFi.h>
#include <PubSubClient.h>

// ==========================================
// PENGATURAN WI-FI & MQTT BROKER
// ==========================================
const char* ssid = "hah";        
const char* password = "123321123"; 
const char* mqtt_server = "test.mosquitto.org"; 
const char* topic_gerak = "robot/gerak"; 
const char* topic_kecepatan = "robot/kecepatan"; // Topik baru untuk kecepatan

WiFiClient espClient;
PubSubClient client(espClient);

// ==========================================
// DEKLARASI PIN MOTOR (REVISI PIN AMAN ESP32)
// ==========================================
const int ENA_B = 13; const int IN1_B = 12; const int IN2_B = 14; // Belakang
const int ENA_L = 25; const int IN1_L = 27; const int IN2_L = 26; // Kiri
const int ENA_R = 4;  const int IN1_R = 2;  const int IN2_R = 15; // Kanan

// ==========================================
// PENGATURAN PWM (KECEPATAN)
// ==========================================
int currentSpeed = 255; // Kecepatan default awal (0 - 255)
const int freq = 30000;
const int resolution = 8;

// ==========================================
// FUNGSI KONEKSI WI-FI
// ==========================================
void setup_wifi() {
  delay(10);
  Serial.println();
  Serial.print("Menghubungkan ke Wi-Fi: ");
  Serial.println(ssid);
  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWi-Fi Terhubung!");
  Serial.print("IP Address: ");
  Serial.println(WiFi.localIP());
}

// ==========================================
// FUNGSI CALLBACK (MENERIMA PERINTAH MQTT)
// ==========================================
void callback(char* topic, byte* message, unsigned int length) {
  String payload;
  for (int i = 0; i < length; i++) {
    payload += (char)message[i];
  }
  
  Serial.print("Topik [");
  Serial.print(topic);
  Serial.print("] Pesan Masuk: ");
  Serial.println(payload);

  // 1. Logika untuk Topik Gerak
  if (String(topic) == topic_gerak) {
    if (payload == "maju") { maju(); }
    else if (payload == "mundur") { mundur(); }
    else if (payload == "geserKanan") { geserKanan(); }
    else if (payload == "geserKiri") { geserKiri(); }
    else if (payload == "serongKiriDepan") { serongKiriDepan(); }
    else if (payload == "serongKananDepan") { serongKananDepan(); }
    else if (payload == "serongKiriBelakang") { serongKiriBelakang(); }
    else if (payload == "serongKananBelakang") { serongKananBelakang(); }
    else if (payload == "putarKanan") { putarKanan(); }
    else if (payload == "putarKiri") { putarKiri(); }
    else if (payload == "berhenti") { berhenti(); }
  } 
  // 2. Logika untuk Topik Kecepatan
  else if (String(topic) == topic_kecepatan) {
    int newSpeed = payload.toInt();
    
    // Pastikan angka yang dimasukkan aman (0 sampai 255)
    if (newSpeed >= 0 && newSpeed <= 255) {
      currentSpeed = newSpeed;
      Serial.print("=> Kecepatan global diubah menjadi: ");
      Serial.println(currentSpeed);
    } else {
      Serial.println("=> Gagal: Masukkan angka antara 0 - 255!");
    }
  }
}

// ==========================================
// FUNGSI RECONNECT MQTT
// ==========================================
void reconnect() {
  while (!client.connected()) {
    Serial.print("Mencoba terhubung ke Broker MQTT...");
    String clientId = "RobotKRSBI-";
    clientId += String(random(0xffff), HEX);
    
    if (client.connect(clientId.c_str())) {
      Serial.println("Terhubung!");
      
      // Subscribe ke DUA topik sekaligus
      client.subscribe(topic_gerak);
      client.subscribe(topic_kecepatan);
      
    } else {
      Serial.print("Gagal, rc=");
      Serial.print(client.state());
      Serial.println(" Coba lagi dalam 5 detik");
      delay(5000);
    }
  }
}

// ==========================================
// SETUP & LOOP
// ==========================================
void setup() {
  Serial.begin(115200);

  // Setup Pin Arah (IN1 & IN2)
  pinMode(IN1_B, OUTPUT); pinMode(IN2_B, OUTPUT);
  pinMode(IN1_L, OUTPUT); pinMode(IN2_L, OUTPUT);
  pinMode(IN1_R, OUTPUT); pinMode(IN2_R, OUTPUT);

  // Setup Pin Kecepatan (ENA) dengan modulasi PWM ESP32
  ledcAttach(ENA_B, freq, resolution);
  ledcAttach(ENA_L, freq, resolution);
  ledcAttach(ENA_R, freq, resolution);

  berhenti(); // Pastikan robot diam saat awal menyala

  setup_wifi();
  client.setServer(mqtt_server, 1883);
  client.setCallback(callback);
}

void loop() {
  if (!client.connected()) {
    reconnect();
  }
  client.loop(); 
}

// ==========================================
// FUNGSI PERGERAKAN OMNI DENGAN KONTROL PWM
// ==========================================

void maju() { // North
  digitalWrite(IN1_L, LOW);  digitalWrite(IN2_L, HIGH); ledcWrite(ENA_L, currentSpeed);
  digitalWrite(IN1_R, LOW);  digitalWrite(IN2_R, HIGH); ledcWrite(ENA_R, currentSpeed); 
  digitalWrite(IN1_B, LOW);  digitalWrite(IN2_B, LOW);  ledcWrite(ENA_B, 0);  // B Mati
}

void mundur() { // South
  digitalWrite(IN1_L, HIGH); digitalWrite(IN2_L, LOW);  ledcWrite(ENA_L, currentSpeed);
  digitalWrite(IN1_R, HIGH); digitalWrite(IN2_R, LOW);  ledcWrite(ENA_R, currentSpeed); 
  digitalWrite(IN1_B, LOW);  digitalWrite(IN2_B, LOW);  ledcWrite(ENA_B, 0);  // B Mati
}

void serongKiriDepan() { // North West
  digitalWrite(IN1_L, LOW);  digitalWrite(IN2_L, LOW);  ledcWrite(ENA_L, 0);  // L Mati
  digitalWrite(IN1_R, LOW);  digitalWrite(IN2_R, HIGH); ledcWrite(ENA_R, currentSpeed); 
  digitalWrite(IN1_B, HIGH); digitalWrite(IN2_B, LOW);  ledcWrite(ENA_B, currentSpeed);
}

void serongKananBelakang() { // South East
  digitalWrite(IN1_L, LOW);  digitalWrite(IN2_L, LOW);  ledcWrite(ENA_L, 0);  // L Mati
  digitalWrite(IN1_R, HIGH); digitalWrite(IN2_R, LOW);  ledcWrite(ENA_R, currentSpeed); 
  digitalWrite(IN1_B, LOW);  digitalWrite(IN2_B, HIGH); ledcWrite(ENA_B, currentSpeed);
}

void serongKananDepan() { // North East
  digitalWrite(IN1_L, LOW);  digitalWrite(IN2_L, HIGH); ledcWrite(ENA_L, currentSpeed); 
  digitalWrite(IN1_R, LOW);  digitalWrite(IN2_R, LOW);  ledcWrite(ENA_R, 0);  // R Mati
  digitalWrite(IN1_B, LOW);  digitalWrite(IN2_B, HIGH); ledcWrite(ENA_B, currentSpeed);
}

void serongKiriBelakang() { // South West
  digitalWrite(IN1_L, HIGH); digitalWrite(IN2_L, LOW);  ledcWrite(ENA_L, currentSpeed); 
  digitalWrite(IN1_R, LOW);  digitalWrite(IN2_R, LOW);  ledcWrite(ENA_R, 0);  // R Mati
  digitalWrite(IN1_B, HIGH); digitalWrite(IN2_B, LOW);  ledcWrite(ENA_B, currentSpeed);
}

void putarKanan() { // Clockwise
  digitalWrite(IN1_L, LOW);  digitalWrite(IN2_L, HIGH); ledcWrite(ENA_L, currentSpeed); 
  digitalWrite(IN1_R, HIGH); digitalWrite(IN2_R, LOW);  ledcWrite(ENA_R, currentSpeed); 
  digitalWrite(IN1_B, HIGH); digitalWrite(IN2_B, LOW);  ledcWrite(ENA_B, currentSpeed);
}

void putarKiri() { // Counter-Clockwise
  digitalWrite(IN1_L, HIGH); digitalWrite(IN2_L, LOW);  ledcWrite(ENA_L, currentSpeed); 
  digitalWrite(IN1_R, LOW);  digitalWrite(IN2_R, HIGH); ledcWrite(ENA_R, currentSpeed); 
  digitalWrite(IN1_B, LOW);  digitalWrite(IN2_B, HIGH); ledcWrite(ENA_B, currentSpeed);
}

// === GESER MURNI (Menggunakan Rumus Fisika 100% Belakang, 50% Depan) ===
void geserKanan() { 
  int halfSpeed = currentSpeed / 2; // Roda depan jalan setengah tenaga
  digitalWrite(IN1_L, LOW);  digitalWrite(IN2_L, HIGH); ledcWrite(ENA_L, halfSpeed);   // L Maju (50%)
  digitalWrite(IN1_R, HIGH); digitalWrite(IN2_R, LOW);  ledcWrite(ENA_R, halfSpeed);   // R Mundur (50%)
  digitalWrite(IN1_B, LOW);  digitalWrite(IN2_B, HIGH); ledcWrite(ENA_B, currentSpeed); // B Dorong Kanan (100%)
}

void geserKiri() { 
  int halfSpeed = currentSpeed / 2; // Roda depan jalan setengah tenaga
  digitalWrite(IN1_L, HIGH); digitalWrite(IN2_L, LOW);  ledcWrite(ENA_L, halfSpeed);   // L Mundur (50%)
  digitalWrite(IN1_R, LOW);  digitalWrite(IN2_R, HIGH); ledcWrite(ENA_R, halfSpeed);   // R Maju (50%)
  digitalWrite(IN1_B, HIGH); digitalWrite(IN2_B, LOW);  ledcWrite(ENA_B, currentSpeed); // B Dorong Kiri (100%)
}

void berhenti() {
  digitalWrite(IN1_L, LOW); digitalWrite(IN2_L, LOW); ledcWrite(ENA_L, 0);
  digitalWrite(IN1_R, LOW); digitalWrite(IN2_R, LOW); ledcWrite(ENA_R, 0);
  digitalWrite(IN1_B, LOW); digitalWrite(IN2_B, LOW); ledcWrite(ENA_B, 0);
}
